LOAD R1, 100         ; Загрузить длину массива в R1
LOAD R2, 99          ; Загрузить базовый адрес массива в R2
LOAD_IND R0, R2      ; Загрузить первый элемент массива в R0
DECREMENT R1, 0      ; Уменьшить R1 (длина массива) на 1
JZ 10, R1            ; Если длина массива стала 0, перейти к завершению
INCREMENT R2, 0      ; Увеличить указатель на следующий элемент
CMP_IND R0, R2       ; Сравнить R0 (текущий максимум) с элементом массива
DECREMENT R1, 0      ; Уменьшить R1 (длина массива) на 1
JZ 10, R1            ; Если длина массива стала 0, перейти к завершению
JUMP 5, 0            ; Перейти к проверке следующего элемента
STORE R0, 150        ; Сохранить найденный максимум в памяти (ячейка 150)
HALT 0, 0            ; Остановить программу
